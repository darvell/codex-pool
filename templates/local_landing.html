Codex CLI Setup
Install
npm install -g @openai/codex
Config (~/.codex/config.toml)
model_provider = "openai"

[model_providers.openai]
name = "OpenAI via Pool"
base_url = "{{.BaseURL}}/v1"
Auth (~/.codex/auth.json)
# Copy any account's auth.json
cp pool/ACCOUNT_NAME/auth.json ~/.codex/auth.json
Claude Code Setup
Install
npm install -g @anthropic-ai/claude-code
Environment Variables
export ANTHROPIC_BASE_URL="{{.BaseURL}}"
export ANTHROPIC_API_KEY="your-api-key"
Gemini CLI Setup
Install
npm install -g @google/gemini-cli
Config (~/.gemini/settings.json)
{
  "codeAssistEndpoint": "{{.BaseURL}}"
}
Auth (~/.gemini/oauth_creds.json)
# Copy Gemini account's auth.json
cp pool/GEMINI_ACCOUNT/auth.json ~/.gemini/oauth_creds.json
